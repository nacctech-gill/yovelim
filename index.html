const action = params.action || 'upload'; if (action === 'finalize') { return handleFinalize(params); } else { return handleUpload(e, params); } } catch (err) { return jsonResponse({ success: false, error: err.message }); } } // ─── העלאת קובץ ────────────────────────────────────────────── function handleUpload(e, params) { const clientName = sanitize(params.clientName || 'לקוח_לא_ידוע'); const docType = sanitize(params.docType || 'general'); const filename = params.filename || (docType + '_' + Date.now()); const mimeType = params.mimeType || 'application/octet-stream'; const fileData = params.fileData; // base64 if (!fileData) { return jsonResponse({ success: true, message: 'no file' }); } // מציאת/יצירת תיקייה ללקוח const clientFolder = getOrCreateFolder(clientName); // פענוח base64 ויצירת קובץ const bytes = Utilities.base64Decode(fileData); const blob = Utilities.newBlob(bytes, mimeType, filename); const savedFile = clientFolder.createFile(blob); // הפיכה לניתן לצפייה savedFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return jsonResponse({ success: true, fileId: savedFile.getId(), fileUrl: savedFile.getUrl(), folder: clientFolder.getUrl() }); } // ─── סיום תהליך ────────────────────────────────────────────── function handleFinalize(params) { const clientName = decodeURIComponent(params.clientName || 'לא ידוע'); const filesList = JSON.parse(params.filesList || '[]'); // שליחת WhatsApp למנהלת sendWhatsApp(clientName, filesList); return jsonResponse({ success: true }); } // ─── שליחת התראה באימייל ───────────────────────────────────── function sendWhatsApp(clientName, filesList) { const lines = filesList.map(f => '• ' + f).join('\n'); const subject = `יובלים גישור - קבצים חדשים מ: ${clientName}`; const body = `שלום,\n\nהלקוח ${clientName} העלה מסמכים חדשים:\n\n${lines}\n\nניתן לצפות בקבצים בגוגל דרייב.\n\nיובלים גישור`; try { GmailApp.sendEmail(MANAGER_EMAIL, subject, body); Logger.log('אימייל נשלח בהצלחה ל-' + MANAGER_EMAIL); } catch (err) { Logger.log('שגיאה בשליחת אימייל: ' + err.message); } } // ─── פונקציות עזר ──────────────────────────────────────────── /** * מחזיר תיקייה קיימת או יוצר חדשה * מבנה: ROOT / שם_לקוח / (קבצים) */ function getOrCreateFolder(clientName) { const root = DriveApp.getFolderById(ROOT_FOLDER_ID); const existing = root.getFoldersByName(clientName); if (existing.hasNext()) { return existing.next(); } return root.createFolder(clientName); } /** * ניקוי קלט - מניעת שמות תיקיות בעייתיים */ function sanitize(str) { return String(str).replace(/[\/\\:*?"<>|]/g, '_').trim().substring(0, 80); } /** * החזרת תגובת JSON עם headers מתאימים */ function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ─── בדיקה ידנית (הרץ מהסקריפט לבדיקה) ────────────────────── function testWhatsApp() { sendWhatsApp('לקוח בדיקה', ['תעודת זהות: test.pdf', 'תעודת נישואין: marriage.pdf']);
